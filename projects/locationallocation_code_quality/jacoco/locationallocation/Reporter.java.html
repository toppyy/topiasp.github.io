package locationallocation;

import locationallocation.Utils.CostMatrix;
import locationallocation.Utils.Location;
import locationallocation.Utils.Writer;

import java.util.Random;
import java.io.IOException;


import java.util.Date;



public class Reporter {

    /**
     * Holds bounds for values of P to be tested.
     */
    private int[] pLimits;

    /**
     * Writes data on performance and quality.
     * @param pLimitsInput Bounds of P to test as [start, end] 
     */
    public Reporter(final int[] pLimitsInput) {

        this.pLimits = pLimitsInput;
        
    }
    /**
     * Runs a single test for given parameters.
     * @param p P.
     * @param candidates Number of candidates.
     * @param demandlocations Number of demandlocations.
     * @param algorithmName Name of algorithm to test.
     * @return String containting test info.
     */
    private String runTest(final int p, final int candidates, final int demandlocations, final String algorithmName) {

        
        double rangeMin = 1;
        double rangeMax = 200;
        
        Location[] testLocations1    = new Location[candidates];
        Location[] testLocations2    = new Location[demandlocations];

        Random r = new Random();
        

        for (int i = 0; i < candidates; i++) {
            double randomValue1 = rangeMin + (rangeMax - rangeMin) * r.nextDouble();
            double randomValue2 = rangeMin + (rangeMax - rangeMin) * r.nextDouble();
            testLocations1[i] = new Location(randomValue1, randomValue2);
        }

        for (int i = 0; i < demandlocations; i++) {
            double randomValue1 = rangeMin + (rangeMax - rangeMin) * r.nextDouble();
            double randomValue2 = rangeMin + (rangeMax - rangeMin) * r.nextDouble();
            testLocations2[i] = new Location(randomValue1, randomValue2);
        }
      
        // Calculate distance matrix
        CostMatrix costs = new CostMatrix(testLocations1, testLocations2);

        Solver a = new Naive(costs, p);
        
        if (algorithmName == "TB") {
            a = new TeitzBart(costs, p);
        } 
        
        if (algorithmName == "GRIA") {
            a = new GRIA(costs, p, testLocations1);
        }
        
        
        long startTime = new Date().getTime();
        a.solve();
        long endTime = new Date().getTime();
		long timeElapsed = endTime - startTime;

        return "" + p + ";" + candidates + ";" + demandlocations + ";" + a.getResultCost() + ";" + timeElapsed + ";" + algorithmName;

    }

    
    /**
     * Runs test for specified algorithm.
     * @param algorithmName  
     */
    public void testAlgorithm(final String algorithmName) {


        int start = this.pLimits[0];
        int end   = this.pLimits[1];

        int candidates = 25;
        int demandlocations = 100;

        int idx = 0;

        String[] results   = new String[end - start];

        for (int i = start; i < end; i++) {
            results[idx]   = this.runTest(i, candidates, demandlocations, algorithmName);
            idx++;
        }

        
       
        try {
            Writer w = new Writer("/home/tob/Desktop/dump/testi_" + algorithmName + ".txt");
           
            for (String s : results) {
                
                w.write(s + "\n");
              
            }

            w.close();

        } catch (IOException e) { 
      
          System.out.println("Problem with writing the results");
          e.printStackTrace(); 
          
          
        } 



        
    }
}
